sudo apt install git - установка систем контроля версий 

git --version - проверка версии 

git confif --global user.email "myemail" -
git confif --global user.name "myname" 

git config --list - проверка настроек 

ssh--keygen - создание ssh ключа 

cat ~ssh/id_rsa.pub - получение ssh ключа 

создаем репозиторий на github

переходим в папку с проектом
git init - превратить текущую директорию в git репозиторий 
git remote add название связи (обычно - origin) ссылка на удаленный репозиторий (SSH: git@github:...)
git remote -v - просмjnh списка удаленных репозиторий 

git remote rm название_связи (обычно origin)- удаление связи с удаленным репозиторием
git status - проверка статуса фацлов
git add название_файла - отслеживание конкретных файлов 
git add . - отслеживанеи всех файлов 

 git comimt - создание коммита (версий /слепка кода)

git commit -m "описание версий" -созданеи коммита с комментарием 
git push название_связи (обычно origin) название_ветки - отправка версии кода на удаленный репозиторий 
напрмиер: git push origin master

git clone ссылка_на_удаленный_дипозиторий - склонировать или скопировать  или скачать удаленный репозиторий

git diff просмотр внесенных изменений 

git branch - вывод списка веток в терминал 
(звездочкой помечается та ветка на которой мы находимся - активная ветка)

git branch название_ветки - создание указанной название_ветки
git checkout название_ветки - переход на указанную ветку

git switch название_ветки - переход на указанную ветку

git pull название_связи название_ветки - стягивание изменений с удаленной репозиторий с указанной название_ветки

git stash  - спрятать локальные данные во временную область

git stash list - просмотр списка спрятанных данных 

git stash apply - запись спрятанных данных/извлечение спрятанных данных

 git stash drop - удаление спрятанных данных 

git log - просмтор коммитов/версий 

gir log --oneline -  компактное отоброжение коммитов
